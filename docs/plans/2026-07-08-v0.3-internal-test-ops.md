export type QualityLevel = 'error' | 'warning';

export interface EventQualityInput {
  id: string;
  eventName?: string | null;
  city?: string | null;
  eventDate?: string | Date | null;
  distanceItems?: string[] | null;
  officialUrl?: string | null;
  sourceName?: string | null;
  sourceLevel?: string | null;
  infoStatus?: string | null;
  runJudgement?: string | null;
  judgementSummary?: string | null;
  judgementReasons?: string[] | null;
  checklistItems?: Array<unknown> | null;
}

export interface EventQualityWarning {
  code: string;
  level: QualityLevel;
  message: string;
}

export function buildEventQualityWarnings(event: EventQualityInput): EventQualityWarning[] {
  const warnings: EventQualityWarning[] = [];

  if (!event.eventName?.trim()) {
    warnings.push({ code: 'missing_event_name', level: 'error', message: '赛事名称不能为空' });
  }
  if (!event.city?.trim()) {
    warnings.push({ code: 'missing_city', level: 'error', message: '城市不能为空' });
  }
  if (!event.eventDate) {
    warnings.push({ code: 'missing_event_date', level: 'error', message: '比赛日期不能为空' });
  }
  if (!event.distanceItems?.length) {
    warnings.push({ code: 'missing_distance', level: 'error', message: '距离项目不能为空' });
  }
  if (!event.officialUrl?.trim()) {
    warnings.push({ code: 'missing_official_url', level: 'error', message: '官方入口不能为空' });
  } else if (event.officialUrl.includes('example.com')) {
    warnings.push({
      code: 'placeholder_official_url',
      level: 'error',
      message: '官方入口不能使用 example.com 占位链接',
    });
  }
  if (!event.sourceName?.trim() || event.sourceLevel === 'unknown') {
    warnings.push({
      code: 'weak_source',
      level: 'warning',
      message: '来源信息仍需人工核实',
    });
  }
  if (event.infoStatus === 'user_flagged') {
    warnings.push({
      code: 'user_flagged',
      level: 'warning',
      message: '赛事存在用户反馈异常，发布前需要处理',
    });
  }
  if (event.runJudgement === 'unverified' || !event.judgementSummary?.trim()) {
    warnings.push({
      code: 'weak_judgement',
      level: 'warning',
      message: '跑前判断摘要或结论仍需补充',
    });
  }
  if (!event.judgementReasons?.length) {
    warnings.push({
      code: 'missing_judgement_reasons',
      level: 'warning',
      message: '判断理由不能为空',
    });
  }
  if (!event.checklistItems?.length) {
    warnings.push({
      code: 'missing_checklist',
      level: 'warning',
      message: '报名前确认清单不能为空',
    });
  }

  return warnings;
}

export function isReleaseReadyEvent(event: EventQualityInput): boolean {
  return !buildEventQualityWarnings(event).some((warning) => warning.level === 'error');
}
```

import { buildEventQualityWarnings, EventQualityInput } from './quality';

type EventInput = EventQualityInput & {
  publishStatus?: string | null;
  eventDate?: string | Date | null;
};

type FeedbackInput = { id: string; status: string; eventId?: string | null };
type ShareRecordInput = { id: string; scene: string; shareType: string };

export function buildOpsSummary(input: {
  events: EventInput[];
  feedback: FeedbackInput[];
  shareRecords: ShareRecordInput[];
}) {
  const pendingFeedback = input.feedback.filter((item) =>
    ['pending', 'handling'].includes(item.status),
  );
  const attentionEvents = input.events
    .map((event) => {
      const warnings = buildEventQualityWarnings(event);
      return {
        id: event.id,
        eventName: event.eventName || '未命名赛事',
        city: event.city || '-',
        eventDate: event.eventDate,
        publishStatus: event.publishStatus || 'draft',
        warnings,
        blockingCount: warnings.filter((warning) => warning.level === 'error').length,
        warningCount: warnings.filter((warning) => warning.level === 'warning').length,
      };
    })
    .filter((event) => event.blockingCount > 0 || event.warningCount > 0)
    .sort((a, b) => b.blockingCount - a.blockingCount || b.warningCount - a.warningCount)
    .slice(0, 10);

  return {
    releaseGate: {
      publishedEvents: input.events.filter((event) => event.publishStatus === 'published').length,
      blockingEvents: attentionEvents.filter((event) => event.blockingCount > 0).length,
      warningEvents: attentionEvents.filter((event) => event.warningCount > 0).length,
      pendingFeedback: pendingFeedback.length,
      userFlaggedEvents: input.events.filter((event) => event.infoStatus === 'user_flagged').length,
    },
    attentionEvents,
    shareSnapshot: {
      total: input.shareRecords.length,
      pageShares: input.shareRecords.filter((record) => record.shareType === 'page_share').length,
      imageGenerates: input.shareRecords.filter((record) => record.shareType === 'image_generate')
        .length,
    },
  };
}
```

export interface ShareStatInput {
  scene: string;
  shareType: string;
  userKey?: string | null;
}

export function buildShareBreakdown(records: ShareStatInput[]) {
  const sceneMap = new Map<string, { scene: string; pageShare: number; imageGenerate: number }>();
  const uniqueUsers = new Set(records.map((record) => record.userKey).filter(Boolean));

  for (const record of records) {
    const entry = sceneMap.get(record.scene) || {
      scene: record.scene,
      pageShare: 0,
      imageGenerate: 0,
    };
    if (record.shareType === 'page_share') entry.pageShare += 1;
    if (record.shareType === 'image_generate') entry.imageGenerate += 1;
    sceneMap.set(record.scene, entry);
  }

  return {
    uniqueUsers: uniqueUsers.size,
    byScene: Array.from(sceneMap.values())
      .map((entry) => ({ ...entry, total: entry.pageShare + entry.imageGenerate }))
      .sort((a, b) => b.total - a.total || a.scene.localeCompare(b.scene)),
  };
}
```

import { readFile } from 'node:fs/promises';
import { resolve } from 'node:path';
import { prisma } from '../src/index.js';

export function checkMiniappConfigText(content: string): string[] {
  const failures: string[] = [];
  if (!content.includes("import { prodConfig } from './prod'")) {
    failures.push('apps/miniapp/config/index.ts must import prodConfig');
  }
  if (!content.includes('export const config = prodConfig')) {
    failures.push('apps/miniapp/config/index.ts must point to prodConfig');
  }
  if (/localhost|127\.0\.0\.1|example\.com|http:\/\//.test(content)) {
    failures.push('apps/miniapp/config/index.ts must not contain local or placeholder API URLs');
  }
  return failures;
}

export function checkProjectConfig(config: unknown): string[] {
  const failures: string[] = [];
  const project = config as { setting?: { urlCheck?: boolean } };
  if (project.setting?.urlCheck !== true) {
    failures.push('apps/miniapp/project.config.json setting.urlCheck must be true');
  }
  return failures;
}

async function main() {
  const root = resolve(process.cwd());
  const failures: string[] = [];
  const indexTs = await readFile(resolve(root, 'apps/miniapp/config/index.ts'), 'utf8');
  failures.push(...checkMiniappConfigText(indexTs));

  const projectConfig = JSON.parse(
    await readFile(resolve(root, 'apps/miniapp/project.config.json'), 'utf8'),
  );
  failures.push(...checkProjectConfig(projectConfig));

  const published = await prisma.event.count({ where: { publishStatus: 'published' } });
  if (published < 5) failures.push('At least 5 published events are required before submission');

  const pendingFeedback = await prisma.feedback.count({
    where: { status: { in: ['pending', 'handling'] } },
  });
  if (pendingFeedback > 0) {
    failures.push(`Pending or handling feedback must be cleared before submission: ${pendingFeedback}`);
  }

  if (failures.length) {
    console.error('release-check failed:');
    failures.forEach((failure) => console.error(`- ${failure}`));
    process.exitCode = 1;
    return;
  }

  console.log('release-check passed');
}

if (process.argv[1]?.endsWith('release-check.ts')) {
  main()
    .catch((error) => {
      console.error(error);
      process.exitCode = 1;
    })
    .finally(async () => {
      await prisma.$disconnect();
    });
}
```
